181 // Factory function to assist in wrapping a callable with signature
182 //
183 // case 1 (keyT != void): void op(const input_keyT&, std::tuple<input_valuesT&...>&&, std::tuple<output_terminalsT...>&)
184 // case 2 (keyT == void): void op(std::tuple<input_valuesT&...>&&, std::tuple<output_terminalsT...>&)
185 template <typename keyT, typename funcT, typename... input_valuesT, typename... output_edgesT>
186 auto wrapt(funcT &&func, const std::tuple<ttg::Edge<keyT, input_valuesT>...> &inedges,
187  const std::tuple<output_edgesT...> &outedges, const std::string &name = "wrapper",
188  const std::vector<std::string> &innames =
189  std::vector<std::string>(std::tuple_size<std::tuple<ttg::Edge<keyT, input_valuesT>...>>::value, "input"),
190  const std::vector<std::string> &outnames =
191  std::vector<std::string>(std::tuple_size<std::tuple<output_edgesT...>>::value, "output")) {
192  using output_terminals_type = typename ttg::edges_to_output_terminals<std::tuple<output_edgesT...>>::type;
193 
194  // Op needs actual types of arguments to func ... extract them and pass to WrapOpArgs
195  // 1. func_args_t = {const input_keyT&, std::tuple<input_valuesT...>&&, std::tuple<output_terminalsT...>&}
196  using func_args_t = boost::callable_traits::args_t<funcT>;
197  constexpr const auto num_args = std::tuple_size<func_args_t>::value;
198  constexpr const auto void_key = ttg::meta::is_void_v<keyT>;
199  static_assert(num_args == (void_key ? 2 : 3), "ttg::wrapt(func, ...): func must take 3 arguments (or 2, if keyT=void)");
200  // 2. input_args_t = {input_valuesT&&...}
201  using input_args_t = std::decay_t<typename std::tuple_element<void_key ? 0 : 1, func_args_t>::type>;
202  using decayed_input_args_t = ttg::meta::decayed_tuple_t<input_args_t>;
203  using wrapT = typename WrapOpUnwrapTuple<funcT, keyT, output_terminals_type, input_args_t>::type;
204  // not sure if we need this level of type checking ...
205  // TODO determine the generic signature of func
206  if constexpr (!void_key) {
207  static_assert(std::is_same_v<typename std::tuple_element<0, func_args_t>::type, const keyT &>,
208  "ttg::wrapt(func, inedges, outedges): first argument of func must be const keyT& (unless keyT = void)");
209  }
210  static_assert(std::is_same_v<decayed_input_args_t, std::tuple<input_valuesT...>>,
211  "ttg::wrapt(func, inedges, outedges): inedges value types do not match argument types of func");
212  static_assert(
213  std::is_same_v<typename std::tuple_element<num_args - 1, func_args_t>::type, output_terminals_type &>,
214  "ttg::wrapt(func, inedges, outedges): last argument of func must be std::tuple<output_terminals_type>&");
215 
216  return std::make_unique<wrapT>(std::forward<funcT>(func), inedges, outedges, name, innames, outnames);
217 }

221 // case 1 (keyT != void): void op(const input_keyT&, input_valuesT&&..., std::tuple<output_terminalsT...>&)
222 // case 2 (keyT == void): void op(input_valuesT&&..., std::tuple<output_terminalsT...>&)
223 template <typename keyT, typename funcT, typename... input_edge_valuesT, typename... output_edgesT>
224 auto wrap(funcT &&func, const std::tuple<ttg::Edge<keyT, input_edge_valuesT>...> &inedges,
225  const std::tuple<output_edgesT...> &outedges, const std::string &name = "wrapper",
226  const std::vector<std::string> &innames = std::vector<std::string>(
227  std::tuple_size<std::tuple<ttg::Edge<keyT, input_edge_valuesT>...>>::value, "input"),
228  const std::vector<std::string> &outnames =
229  std::vector<std::string>(std::tuple_size<std::tuple<output_edgesT...>>::value, "output")) {
230  using output_terminals_type = typename ttg::edges_to_output_terminals<std::tuple<output_edgesT...>>::type;
231 
232  // Op needs actual types of arguments to func ... extract them and pass to WrapOpArgs
233  // 1. func_args_t = {const input_keyT&, input_valuesT&&..., std::tuple<output_terminalsT...>&}
234  using func_args_t = boost::callable_traits::args_t<funcT>;
235  constexpr const auto num_args = std::tuple_size<func_args_t>::value;
236  constexpr const auto void_key = ttg::meta::is_void_v<keyT>;
237  static_assert(num_args == sizeof...(input_edge_valuesT) + (void_key ? 1 : 2),
238  "ttg::wrap(func, inedges): func's # of args != # of inedges");
239  // 2. input_args_t = {input_valuesT&&...}
240  using input_args_t = typename ttg::meta::take_first_n<
241  typename ttg::meta::drop_first_n<func_args_t, std::size_t(void_key ? 0 : 1)>::type,
242  std::tuple_size<func_args_t>::value - (void_key ? 1 : 2)>::type;
243  using decayed_input_args_t = ttg::meta::decayed_tuple_t<input_args_t>;
244  using wrapT = typename WrapOpArgsUnwrapTuple<funcT, keyT, output_terminals_type, input_args_t>::type;
245  // not sure if we need this level of type checking ...
246  // TODO determine the generic signature of func
247  if constexpr (!void_key) {
248  static_assert(std::is_same_v<typename std::tuple_element<0, func_args_t>::type, const keyT &>,
249  "ttg::wrap(func, inedges, outedges): first argument of func must be const keyT& (unless keyT = void)");
250  }
251  static_assert(std::is_same_v<decayed_input_args_t, std::tuple<input_edge_valuesT...>>,
252  "ttg::wrap(func, inedges, outedges): inedges value types do not match argument types of func");
253  static_assert(
254  std::is_same_v<typename std::tuple_element<num_args - 1, func_args_t>::type, output_terminals_type &>,
255  "ttg::wrap(func, inedges, outedges): last argument of func must be std::tuple<output_terminals_type>&");
256 
257  return std::make_unique<wrapT>(std::forward<funcT>(func), inedges, outedges, name, innames, outnames);
258 }
