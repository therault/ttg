31  if (!op) {

42  int count = 0;
43  for (auto in : op->get_inputs()) {
44  if (!in) {

46  status = false;
47  } else {

49  if (!in->is_connected()) {
50  std::cout << "ttg::Traverse: " << op->get_name() << " input terminal #" << count << " " << in->get_name() << " is not connected\n";
51  status = false;
52  }
53  }
54  count++;
55  }
56 
57  count = 0;
58  for (auto out : op->get_outputs()) {
59  if (!out) {

61  status = false;
62  } else {

64  if (!out->is_connected()) {
65  std::cout << "ttg::Traverse: " << op->get_name() << " output terminal #" << count << " " << out->get_name() << " is not connected\n";
66  status = false;
67  }
68  }
69  count++;
70  }

72  for (auto out : op->get_outputs()) {

75  if (!successor) {
