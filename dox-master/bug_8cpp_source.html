106 void Debugger::handle(int sig) {
107  if (sig >= NSIG) return;
108  typedef void (*handler_type)(int);

200 void Debugger::debug(const char *reason) {
201  std::cout << prefix_ << "Debugger::debug: ";
202  if (reason)
203  std::cout << reason;
204  else
205  std::cout << "no reason given";
206  std::cout << endl;
207 
208  if (!cmd_.empty()) {
209  int pid = getpid();
210  // contruct the command name
211  std::string cmd = cmd_;
212  std::string::size_type pos;
213  std::string pidvar("$(PID)");
214  while ((pos = cmd.find(pidvar)) != std::string::npos) {
215  std::string pidstr;
216  pidstr += std::to_string(pid);
217  cmd.replace(pos, pidvar.size(), pidstr);
218  }
219  std::string execvar("$(EXEC)");
220  while ((pos = cmd.find(execvar)) != std::string::npos) {
221  cmd.replace(pos, execvar.size(), exec_);
222  }
223  std::string prefixvar("$(PREFIX)");
224  while ((pos = cmd.find(prefixvar)) != std::string::npos) {
225  cmd.replace(pos, prefixvar.size(), prefix_);
226  }
227 
228  // start the debugger
229  // before starting the debugger de-register signal handler for SIGTRAP to let the debugger take over
230  release(SIGTRAP);
231  std::cout << prefix_ << "Debugger: starting \"" << cmd << "\"" << endl;
232  debugger_ready_ = 0;
233  const auto system_retvalue = system(cmd.c_str());
234  if (system_retvalue != 0) { // call to system() failed
235  std::cout << prefix_ << "Failed debugger launch: system() did not succeed ..." << endl;
236  } else { // call to system() succeeded
237  // wait until the debugger is ready
238  if (sleep_) {
239  std::cout << prefix_ << "Sleeping " << sleep_ << " seconds to wait for debugger ..." << endl;
240  sleep(sleep_);
241  }
242  if (wait_for_debugger_) {
243  std::string make_ready_message;
244  if (cmd_.find(" gdb ") != std::string::npos || cmd_.find(" lldb ") != std::string::npos) {
245  make_ready_message = " configure debugging session (set breakpoints/watchpoints, etc.) then type 'c' to continue running";
246  }
247 
248  std::cout << prefix_
249  << ": waiting for the user ..." << make_ready_message << endl;
250  while (!debugger_ready_)
251  ;
252  }
253  }
254  }

272 #endif

274  signame = "SIGTRAP";
275  else
276  signame = "UNKNOWN SIGNAL";
277 
278  if (traceback_) {
279  traceback(signame);
280  }
281  if (debug_) {
282  debug(signame);
283  }
284 
285  if (exit_on_signal_) {
286  std::cout << prefix_ << "Debugger: exiting" << endl;
287  exit(1);
288  } else {
289  std::cout << prefix_ << "Debugger: continuing" << endl;
290  }

311 void Debugger::__traceback(const std::string &prefix, const char *reason) {
312  detail::Backtrace result(prefix);
313  const size_t nframes_to_skip = 2;
314 #if defined(HAVE_LIBUNWIND)
315  std::cout << prefix << "Debugger::traceback(using libunwind):";
316 #elif defined(HAVE_BACKTRACE) // !HAVE_LIBUNWIND
317  std::cout << prefix << "Debugger::traceback(using backtrace):";
318 #else // !HAVE_LIBUNWIND && !HAVE_BACKTRACE
319 #if defined(SIMPLE_STACK)
320  std::cout << prefix << "Debugger::traceback:";
321 #else
322  std::cout << prefix << "traceback not available for this arch" << endl;
323  return;
324 #endif // SIMPLE_STACK
325 #endif // HAVE_LIBUNWIND, HAVE_BACKTRACE
326 
327  if (reason)
328  std::cout << reason;
329  else
330  std::cout << "no reason given";
331  std::cout << endl;
332 
333  if (result.empty())
334  std::cout << prefix << "backtrace returned no state information" << std::endl;
335  else
336  std::cout << result.str(nframes_to_skip) << std::endl;
337 }
