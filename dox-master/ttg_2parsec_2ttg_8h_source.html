96  int src_rank, void *obj) {

98  parsec_taskpool_t *tp = NULL;
99  msg_header_t *msg = static_cast<msg_header_t *>(data);
100  uint64_t op_id = msg->op_id;
101  tp = parsec_taskpool_lookup(msg->taskpool_id);
102  assert(NULL != tp);
103  static_map_mutex.lock();
104  try {
105  auto op_pair = static_id_to_op_map.at(op_id);
106  static_map_mutex.unlock();
107  tp->tdm.module->incoming_message_start(tp, src_rank, NULL, NULL, 0, NULL);
108  static_set_arg_fct = op_pair.first;
109  static_set_arg_fct(data, size, op_pair.second);
110  tp->tdm.module->incoming_message_end(tp, NULL);
111  return 0;
112  } catch (const std::out_of_range &e) {
113  void *data_cpy = malloc(size);
114  assert(data_cpy != 0);
115  memcpy(data_cpy, data, size);
116  if (ttg::tracing()) {
117  ttg::print("ttg_parsec(", ttg_default_execution_context().rank(), ") Delaying delivery of message (",
118  src_rank, ", ", op_id, ", ", data_cpy, ", ", size, ")");
119  }
120  delayed_unpack_actions.insert(std::make_pair(op_id, std::make_tuple(src_rank, data_cpy, size)));

488  inline uint64_t parsec_tasks_hash_fct(parsec_key_t key, int nb_bits, void *data) {
489  /* Use all the bits of the 64 bits key, project on the lowest base bits (0 <= hash < 1024) */
490  int b = 0;
491  uint64_t mask = ~0ULL >> (64 - nb_bits);
492  uint64_t h = (uint64_t)key;
493  (void)data;
494  while (b < 64) {
495  b += nb_bits;
496  h ^= (uint64_t)key >> b;
497  }
498  return (uint64_t)(h & mask);
499  }

563  int32_t readers = parsec_atomic_fetch_dec_int32(&copy->readers);
564  if (1 == readers) {
565  copy->delete_fn(copy->device_private);
566  copy->device_private = NULL;
567  }
568  }
569  }
570  if (NULL != copy->push_task) {
571  /* Release the task if it was deferrred */
572  parsec_task_t *push_task = copy->push_task;
573  if (parsec_atomic_cas_ptr(&copy->push_task, push_task, nullptr)) {
574  parsec_ttg_task_base_t *deferred_op = (parsec_ttg_task_base_t *)copy->push_task;
575  assert(deferred_op->deferred_release);
576  deferred_op->deferred_release(deferred_op->op_ptr, deferred_op);
577  }
578  }
579  PARSEC_OBJ_RELEASE(copy);
580  }
581  }

583  template <typename Value>
584  inline ttg_data_copy_t *register_data_copy(ttg_data_copy_t *copy_in, parsec_ttg_task_base_t *task, bool readonly) {
585  ttg_data_copy_t *copy_res = copy_in;
586  bool replace = false;
587  int32_t readers = -1;
588  if (readonly && copy_in->readers > 0) {
589  /* simply increment the number of readers */
590  readers = parsec_atomic_fetch_inc_int32(&copy_in->readers);
591  }
592  if (readers < 0) {
593  /* someone is going to write into this copy -> we need to make a copy */
594  copy_res = NULL;
595  if (readonly) {
596  replace = true;
597  }
598  } else if (!readonly) {
599  /* this task will mutate the data
600  * check whether there are other readers already and potentially
601  * defer the release of this task to give following readers a
602  * chance to make a copy of the data before this task mutates it
603  *
604  * Try to replace the readers with a negative value that indicates
605  * the value is mutable. If that fails we know that there are other
606  * readers or writers already.
607  */
608  if (parsec_atomic_cas_int32(&copy_in->readers, 1, INT32_MIN)) {
613  assert(nullptr == copy_in->push_task);
614  assert(nullptr != task);
615  copy_in->push_task = &task->parsec_task;
616  } else {
617  /* there are readers of this copy already, make a copy that we can mutate */
618  copy_res = NULL;
619  }
620  }
621  if (NULL != copy_res) {
622  PARSEC_OBJ_RETAIN(copy_res);
623  }
624 
625  if (NULL == copy_res) {
626  ttg_data_copy_t *new_copy = detail::create_new_datacopy(*static_cast<Value *>(copy_in->device_private));
627  if (replace) {
628  /* TODO: Make sure there is no race condition with the release in release_data_copy,
629  * in particular when it comes to setting the callback and replacing the data */
630 
631  /* replace the task that was deferred */
632  parsec_ttg_task_base_t *deferred_op = (parsec_ttg_task_base_t *)copy_in->push_task;
633  ttg_data_copy_t *deferred_replace_copy;
634  deferred_replace_copy = detail::find_copy_in_task(copy_in->push_task, copy_in->device_private);
635  /* replace the copy in the deferred task */
636  for (int i = 0; i < MAX_PARAM_COUNT; ++i) {
637  if (copy_in->push_task->data[i].data_in == copy_in) {
638  copy_in->push_task->data[i].data_in = new_copy;
639  break;
640  }
641  }
642  assert(deferred_op->deferred_release);
643  deferred_op->deferred_release(deferred_op->op_ptr, deferred_op);
644  copy_in->push_task = NULL;
645  copy_in->readers = 1; // set the copy back to being read-only
646  ++copy_in->readers; // register as reader
647  copy_res = copy_in; // return the copy we were passed
648  } else {
649  copy_res = new_copy; // return the new copy
650  }
651  }
652  return copy_res;
653  }
