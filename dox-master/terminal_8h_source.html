288  std::enable_if_t<meta::is_none_void_v<Key,Value> && std::is_same_v<Value,std::remove_reference_t<Value>>,void>
289  send(const Key &key, Value &&value) {
290  std::size_t N = successors().size();
291  // find the first terminal that can consume the value
292  std::size_t move_terminal = N - 1;
293  for (std::size_t i = 0; i != N; ++i) {
294  if (successors().at(i)->get_type() == TerminalBase::Type::Consume) {
295  move_terminal = i;
296  break;
297  }
298  }
299  if (N > 0) {
300  // send copies to every terminal except the one we will move the results to
301  for (std::size_t i = 0; i != N; ++i) {
302  if (i != move_terminal) {
303  TerminalBase *successor = successors().at(i);
304  if (successor->get_type() == TerminalBase::Type::Read) {
305  static_cast<In<keyT, std::add_const_t<valueT>> *>(successor)->send(key, value);
306  } else if (successor->get_type() == TerminalBase::Type::Consume) {
307  static_cast<In<keyT, valueT> *>(successor)->send(key, value);
308  }
309  }
310  }
311  {
312  TerminalBase *successor = successors().at(move_terminal);
313  static_cast<In<keyT, valueT> *>(successor)->send(key, std::forward<Value>(value));
314  }
315  }
