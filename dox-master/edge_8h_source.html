15  template <typename keyT, typename valueT>
16  class Edge {
17  private:
18  // An EdgeImpl represents a single edge that most usually will
19  // connect a single output terminal with a single
20  // input terminal. However, we had to relax this constraint in
21  // order to easily accommodate connecting an input/output edge to
22  // an operation that to the outside looked like a single op but
23  // internally was implemented as multiple operations. Thus, the
24  // input/output edge has to connect to multiple terminals.
25  // Permitting multiple end points makes this much easier to
26  // compose, easier to implement, and likely more efficient at
27  // runtime. This is why outs/ins are vectors rather than pointers
28  // to a single terminal.
29  struct EdgeImpl {
30  std::string name;
31  std::vector<TerminalBase *> outs; // In<keyT, valueT> or In<keyT, const valueT>
32  std::vector<Out<keyT, valueT> *> ins;
33 
34  EdgeImpl() : name(""), outs(), ins() {}
35 
36  EdgeImpl(const std::string &name) : name(name), outs(), ins() {}
37 
38  void set_in(Out<keyT, valueT> *in) {
39  if (ins.size() && tracing()) {
40  print("Edge: ", name, " : has multiple inputs");
41  }
42  ins.push_back(in);
43  try_to_connect_new_in(in);
44  }
45 
46  void set_out(TerminalBase *out) {
47  if (outs.size() && tracing()) {
48  print("Edge: ", name, " : has multiple outputs");
49  }
50  outs.push_back(out);
51  try_to_connect_new_out(out);
52  }
53 
54  void try_to_connect_new_in(Out<keyT, valueT> *in) const {
55  for (auto out : outs)
56  if (in && out) in->connect(out);
57  }
58 
59  void try_to_connect_new_out(TerminalBase *out) const {
60  assert(out->get_type() != TerminalBase::Type::Write); // out must be an In<>
61  for (auto in : ins)
62  if (in && out) in->connect(out);
63  }
64 
65  ~EdgeImpl() {
66  if (ins.size() == 0 || outs.size() == 0) {
67  std::cerr << "Edge: destroying edge pimpl ('" << name << "') with either in or out not "
68  "assigned --- graph may be incomplete"
69  << std::endl;
70  }
71  }
72  };
